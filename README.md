# THE-BEGINNING.v1
Knowledge based 🧠 Core Idea
The central question is:
“If civilization collapsed and you were sent back to the beginning of human history with only your knowledge, what would you need to rebuild the world?”
It explores the chain of human progress:
Nature → Tools → Materials → Machines → Energy → Society → Culture
The genius of the concept is that it forces you to realize something:
A smartphone looks like magic… but it is actually the final layer of thousands of tiny discoveries:
	•	fire
	•	metallurgy
	•	glass
	•	optics
	•	electricity
	•	chemistry
	•	manufacturing
	•	mathematics
	•	communication
The book tries to reveal those hidden foundations. 

📚 The “Civilization Stack” Analysis
1. Survival Layer 🌱
The first civilization isn’t built with computers.
It starts with:
	•	finding clean water
	•	preserving food
	•	farming
	•	animal domestication
	•	shelter
	•	basic medicine
The deeper message:
A civilization is not a collection of inventions — it is a system of dependencies.
A simple example:
A hammer requires:
	•	metal
	•	mining
	•	heat
	•	shaping tools
	•	transportation
One object contains an entire civilization inside it.

2. Technology Tree ⚙️
The book feels almost like a real-life version of a video game tech tree.
You don’t unlock:
“Electricity”
You unlock:
	1	copper mining
	2	wire production
	3	magnets
	4	generators
	5	batteries
	6	circuits
Humanity advanced because every generation stood on previous layers.

3. The Philosophy Behind It 🌌
The most interesting part isn’t the “how-to.”
It’s the philosophical question:
What knowledge is actually essential?
Modern people carry enormous knowledge in society, but individually we know very little.
A person may use:
	•	a car
	•	a phone
	•	medicine
	•	electricity
…but couldn’t recreate them from raw materials.
The book exposes how fragile and incredible civilization is.

🔥 The Dark Side / Criticism
A realistic critique:
It is not a magic “restart civilization” manual.
Rebuilding a world would require:
	•	thousands of specialists
	•	massive labor
	•	resources
	•	institutions
	•	cooperation
Knowing how a steam engine works is different from having:
	•	iron mines
	•	factories
	•	skilled workers
	•	supply chains
Some readers view it more as a beautiful encyclopedia of human invention than a true survival guide. 

The Deeper Theme ❤️
To me, the most powerful idea is:
Civilization is memory.
Every bridge, book, machine, and city is humanity storing knowledge outside the human mind.
A library burning is not just paper disappearing.
It is thousands of years of human experiments disappearing

// === INPUT HANDLING (Professional abstraction) ===
function setupInput() {
 // Desktop keyboard
 document.addEventListener('keydown', e => {
 keys[e.code] = true;
 if (e.code === 'KeyB') toggleBuildMode();
 if (e.code === 'Escape') toggleMenu();
 if (e.code === 'KeyE') tryInteract();
 });
 document.addEventListener('keyup', e => keys[e.code] = false);

 // Pointer lock for desktop look
 canvas.addEventListener('click', () => {
 if (!isMobile && document.pointerLockElement !== canvas) {
 canvas.requestPointerLock();
 }
 });
 document.addEventListener('pointerlockchange', () => {
 if (document.pointerLockElement === canvas) {
 document.addEventListener('mousemove', onMouseMove, false);
 } else {
 document.removeEventListener('mousemove', onMouseMove, false);
 }
 });

 function onMouseMove(e) {
 if (document.pointerLockElement === canvas) {
 controls.yaw -= e.movementX * 0.002;
 controls.pitch -= e.movementY * 0.002;
 controls.pitch = Math.max(-Math.PI/2 + 0.1, Math.min(Math.PI/2 - 0.1, controls.pitch));
 }
 }

 // === MOBILE TOUCH CONTROLS (Landscape optimized) ===
 if (isMobile) {
 const joystick = document.getElementById('joystick');
 const knob = document.getElementById('joystick-knob');
 const jumpBtn = document.getElementById('jump-btn');
 const interactBtn = document.getElementById('interact-btn');
 const buildBtn = document.getElementById('build-btn');

 let joystickActive = false;
 let joystickTouchId = null;
 let lookTouchId = null;
 let lastLookX = 0, lastLookY = 0;

 // Joystick movement
 joystick.addEventListener('touchstart', e => {
 e.preventDefault();
 joystickActive = true;
 joystickTouchId = e.changedTouches[0].identifier;
 updateJoystick(e.changedTouches[0]);
 }, {passive: false});

 joystick.addEventListener('touchmove', e => {
 e.preventDefault();
 for (let t of e.changedTouches) {
 if (t.identifier === joystickTouchId) updateJoystick(t);
 }
 }, {passive: false});

 joystick.addEventListener('touchend', e => {
 e.preventDefault();
 joystickActive = false;
 input.forward = 0;
 input.strafe = 0;
 knob.style.transform = 'translate(-50%, -50%)';
 });

 function updateJoystick(touch) {
 const rect = joystick.getBoundingClientRect();
 const centerX = rect.left + rect.width/2;
 const centerY = rect.top + rect.height/2;
 let dx = (touch.clientX - centerX) / (rect.width/2);
 let dy = (touch.clientY - centerY) / (rect.height/2);
 
 const dist = Math.min(1, Math.sqrt(dx*dx + dy*dy));
 if (dist > 0) {
 dx /= dist; dy /= dist; // normalize
 dx *= Math.min(1, dist);
 dy *= Math.min(1, dist);
 }
 
 input.strafe = dx;
 input.forward = -dy; // forward is negative Y on screen
 
 knob.style.transform = `translate(calc(-50% + ${dx*40}px), calc(-50% + ${dy*40}px))`;
 }

 // Look / Camera drag on right side
 canvas.addEventListener('touchstart', e => {
 for (let t of e.changedTouches) {
 if (t.clientX > window.innerWidth * 0.45 && !joystickActive) {
 lookTouchId = t.identifier;
 lastLookX = t.clientX;
 lastLookY = t.clientY;
 }
 }
 }, {passive: false});

 canvas.addEventListener('touchmove', e => {
 for (let t of e.changedTouches) {
 if (t.identifier === lookTouchId) {
 const dx = t.clientX - lastLookX;
 const dy = t.clientY - lastLookY;
 controls.yaw -= dx * 0.0035;
 controls.pitch -= dy * 0.0035;
 controls.pitch = Math.max(-1.4, Math.min(1.0, controls.pitch));
 lastLookX = t.clientX;
 lastLookY = t.clientY;
 }
 }
 }, {passive: false});

 canvas.addEventListener('touchend', e => {
 for (let t of e.changedTouches) {
 if (t.identifier === lookTouchId) lookTouchId = null;
 }
 });

 // Action buttons
 jumpBtn.addEventListener('touchstart', e => { e.preventDefault(); input.jump = true; });
 jumpBtn.addEventListener('touchend', () => input.jump = false);

 interactBtn.addEventListener('touchstart', e => { e.preventDefault(); tryInteract(); });
 buildBtn.addEventListener('touchstart', e => { e.preventDefault(); toggleBuildMode(); });
 }

 // Desktop build & interact
 window.addEventListener('keydown', e => {
 if (e.code === 'Space') input.jump = true;
 if (e.code === 'KeyE') tryInteract();
 });
 window.addEventListener('keyup', e => {
 if (e.code === 'Space') input.jump = false;
 });
}

 camera.position.y += 1.7;

 // === SKY & LIGHTING (Golden hour sunset) ===
 renderer.setClearColor(0x1f160f);
 
 const hemiLight = new THREE.HemisphereLight(0xffcc88, 0x334455, 0.6);
 scene.add(hemiLight);

 const sunLight = new THREE.DirectionalLight(0xffaa66, 1.1);
 sunLight.position.set(120, 180, -80);
 sunLight.castShadow = quality >= 1;
 if (quality >= 1) {
 sunLight.shadow.mapSize.width = quality === 2 ? 2048 : 1024;
 sunLight.shadow.mapSize.height = sunLight.shadow.mapSize.width;
 sunLight.shadow.camera.near = 10;
 sunLight.shadow.camera.far = 400;
 sunLight.shadow.camera.left = -180;
 sunLight.shadow.camera.right = 180;
 sunLight.shadow.camera.top = 180;
 sunLight.shadow.camera.bottom = -180;
 }
 scene.add(sunLight);

 // === GROUND & TERRAIN ===
 const groundGeo = new THREE.PlaneGeometry(600, 600, 80, 80);
 const groundMat = new THREE.MeshLambertMaterial({ 
 color: 0x3a2f22, 
 flatShading: quality < 2 
 });
 const ground = new THREE.Mesh(groundGeo, groundMat);
 ground.rotation.x = -Math.PI / 2;
 ground.receiveShadow = true;
 scene.add(ground);
